7) display Pascal's triangle

#include <stdio.h>
int main() {
   int n, coef = 1, i, j;
   printf("Enter the number of rows: ");
   scanf("%d", &n);
   for (i = 0; i < n; i++) {
      for (j = 1; j <= n - i; j++)
         printf("  ");
      for (j = 0; j <= i; j++) {
         if (j == 0 || i == 0)
            coef = 1;
         else
            coef = coef * (i - j + 1) / j;
         printf("%4d", coef);
      }
      printf("\n");
   }
   return 0;
}

10) find the Armstrong number for a given range of number.

#include<stdio.h>
int main()
{
    int i, temp, start, end, sum, rem;
    printf("enter a starting range: \n");
    scanf("%d", &start);
    printf("enter a ending range: \n");
    scanf("%d", &end);
    printf("armstrong numbers are: \n");
    for(i=start; i<end; i++){
        sum=0;
        temp=i;
        while(temp != 0){
            rem = temp % 10;
            sum += rem*rem*rem;
            temp = temp / 10;
        }
        if(sum == i)
            printf("%d  ", i);
        
    }
    return 0;
}
